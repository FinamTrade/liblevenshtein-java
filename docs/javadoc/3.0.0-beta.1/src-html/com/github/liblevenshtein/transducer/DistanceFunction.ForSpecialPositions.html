001package com.github.liblevenshtein.transducer;
002
003import java.io.Serializable;
004
005/**
006 * The distance of each position in a state can be defined as follows:
007 * <p>
008 * <code>distance = w - i + e</code>
009 * <p>
010 * For every accepting position, it must be the case that <code>w - i &le; n - e</code>.
011 * It follows directly that the distance of every accepted position must be no
012 * more than {@code n}:
013 * <p>
014 * <code>(w - i &le; n - e) &hArr; (w - i + e &le; n) &hArr; (distance &le; n)</code>
015 * <p>
016 * The Levenshtein distance between any two terms is defined as the minimum
017 * edit distance between the two terms.  Therefore, iterate over each position
018 * in an accepting state, and take the minimum distance among all its accepting
019 * positions as the corresponding Levenshtein distance.
020 * @author Dylon Edwards
021 * @since 2.1.0
022 */
023public abstract class DistanceFunction implements Serializable {
024
025  private static final long serialVersionUID = 1L;
026
027  /**
028   * @param state Levenshtein state whose minimum, Levenshtein distance is to be
029   * determined.
030   * @param w Length of the query term.
031   * @return Minimum distance between the spelling candidate and query term,
032   * implied by the positions within {@code state}.
033   */
034  public abstract int at(State state, int w);
035
036  /**
037    * Distance function for the standard, Levenshtein distance algorithm.
038    * @author Dylon Edwards
039    * @since 2.1.0
040    */
041  public static class ForStandardPositions extends DistanceFunction {
042
043    private static final long serialVersionUID = 1L;
044
045    /**
046      * {@inheritDoc}
047      */
048    @Override
049    public int at(final State state, final int queryLength) {
050      int minimumDistance = Integer.MAX_VALUE;
051
052      for (final Position position : state) {
053        final int i = position.termIndex();
054        final int e = position.numErrors();
055        final int distance = queryLength - i + e;
056        if (distance < minimumDistance) {
057          minimumDistance = distance;
058        }
059      }
060
061      return minimumDistance;
062    }
063  }
064
065  /**
066    * Distance function for the Levenshtein distance algorithms, extended with
067    * transposition (standard- and t-positions), and with merge and split
068    * (standard- and s-positions).
069    * @author Dylon Edwards
070    * @since 2.1.0
071    */
072  public static class ForSpecialPositions extends DistanceFunction {
073
074    private static final long serialVersionUID = 1L;
075
076    /**
077      * {@inheritDoc}
078      */
079    @Override
080    public int at(final State state, final int queryLength) {
081      int minimumDistance = Integer.MAX_VALUE;
082
083      for (final Position position : state) {
084        if (!position.isSpecial()) {
085          final int i = position.termIndex();
086          final int e = position.numErrors();
087          final int distance = queryLength - i + e;
088          if (distance < minimumDistance) {
089            minimumDistance = distance;
090          }
091        }
092      }
093
094      return minimumDistance;
095    }
096  }
097}
