099  public synchronized boolean add(@NonNull final String term) {
100    if (term == null) {
101      throw new java.lang.IllegalArgumentException("term is null");
102    }
103    if (term.compareTo(previousTerm) < 0) {
104      throw new IllegalArgumentException("Due to caveats with the current DAWG implementation, terms must be inserted in ascending order");
105    }
106    // Special Case: Empty String
107    if (term.isEmpty()) {
108      root = new FinalDawgNode();
109      return true;
110    }
111    final int upperBound = term.length() < previousTerm.length() ? term.length() : previousTerm.length();
112    // Find the length of the longest, common prefix between term and
113    // previousTerm
114    int i = 0;
115    while (i < upperBound && term.charAt(i) == previousTerm.charAt(i)) {
116      i += 1;
117    }
118    // Check the unchecked nodes for redundancy, proceeding from the last one
119    // down to the common prefix size. Then, truncate the list at that point.
120    minimize(i);
121    // Add the suffix, starting from the correct node, mid-way through the graph
122    DawgNode node = (null == uncheckedTransitions.peekFirst()) ? root : uncheckedTransitions.peekFirst().target();
123    for (int k = term.length() - 1; i < k; i += 1) {
124      final char label = term.charAt(i);
125      final DawgNode nextNode = new DawgNode();
126      uncheckedTransitions.addFirst(new Transition(node, label, nextNode));
127      node = nextNode;
128    }
129    if (i < term.length()) {
130      final char label = term.charAt(i);
131      final DawgNode nextNode = new FinalDawgNode();
132      uncheckedTransitions.addFirst(new Transition(node, label, nextNode));
133    }
134    //node = nextNode;
135    //i += 1;
136    previousTerm = term;
137    size += 1;
138    return true;
139  }

148  /**
149   * Builds this DAWG in such a way that it remains a minimal trie.
150   * @param lowerBound Number of pending transitions to leave for the next
151   * round (they will be the most-recent transitions).
152   */
153  private void minimize(final int lowerBound) {
154    // Proceed from the leaf up to a certain point
155    for (int j = uncheckedTransitions.size(); j > lowerBound; j -= 1) {
156      final Transition transition = uncheckedTransitions.removeFirst();
157      final DawgNode source = transition.source();
158      final char label = transition.label();
159      final DawgNode target = transition.target();
160      final DawgNode existing = minimizedNodes.get(target);
161      if (null != existing) {
162        source.addEdge(label, existing);
163      } else {
164        source.addEdge(label, target);
165        minimizedNodes.put(target, target);
166      }
167    }
