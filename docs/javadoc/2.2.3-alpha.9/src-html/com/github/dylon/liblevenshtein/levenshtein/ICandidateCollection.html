001package com.github.dylon.liblevenshtein.levenshtein;
002
003/**
004 * Describes the operations necessary for all collections that hold spelling
005 * candidates for query terms.
006 * @param <Type> Kind of the spelling candidates of this collection.
007 * @author Dylon Edwards
008 * @since 2.1.0
009 */
010public interface ICandidateCollection<Type> extends Iterable<Type> {
011
012  /**
013   * Adds a spelling candidate to this collection if it may be added.  Some
014   * reasons it might not be added include if the collection has reached its
015   * maximum size, or if the collection encapsulates a heap of candidates with
016   * the shortest distance, whether the collection has reached its maximum size
017   * and the term's distance is too great to replace one of the terms within.
018   * @param term Spelling candidate to offer to this collection.
019   * @param distance Levenshtein distance of the spelling candidate with the
020   * query term.
021   * @return Whether {@code term} was accepted into this collection.
022   */
023  default boolean offer(String term, int distance) {
024    throw new UnsupportedOperationException(
025      "offer(String,int) is not supported by this implementation");
026  }
027}
