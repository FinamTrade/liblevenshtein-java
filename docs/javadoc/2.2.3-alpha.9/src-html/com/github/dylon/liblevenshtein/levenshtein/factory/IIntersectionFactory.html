001package com.github.dylon.liblevenshtein.levenshtein.factory;
002
003import java.io.Serializable;
004
005import com.github.dylon.liblevenshtein.levenshtein.Intersection;
006import com.github.dylon.liblevenshtein.levenshtein.IState;
007
008/**
009 * Builds nodes that help intersect a dictionary automaton with a Levenshtein
010 * automaton.
011 * @param <DictionaryNode> Kind of the nodes in the dictionary automaton.
012 * @author Dylon Edwards
013 * @since 2.1.0
014 */
015public interface IIntersectionFactory<DictionaryNode> extends Serializable {
016
017  long serialVersionUID = 1L;
018
019  /**
020   * Constructs a new Intersection between two states: one from the dictionary
021   * and one from the Levenshtein automaton.
022   * @param candidate Substring of the candidate term from the dictionary
023   * (spelling candidate of the query term).
024   * @param dictionaryNode Dictionary node representing the final character in
025   * the candidate term.
026   * @param levenshteinState Location within the Levenshtein automaton of this
027   * intersection.
028   * @return New intersection node between {@code dictionaryNode} and
029   * {@code levenshteinState}, having the term-prefix, {@code candidate}.
030   */
031  Intersection<DictionaryNode> build(
032      String candidate,
033      DictionaryNode dictionaryNode,
034      IState levenshteinState);
035
036  /**
037   * [Optional Operation] Instructs this factory that the intersection will no
038   * longer be used and may be recycled for reuse.
039   * @param intersection Value to recycle.
040   */
041  default void recycle(final Intersection<DictionaryNode> intersection) {
042    // default behavior is to do nothing
043  }
044}
