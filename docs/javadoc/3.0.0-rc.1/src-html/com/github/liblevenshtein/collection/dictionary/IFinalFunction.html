001package com.github.liblevenshtein.collection.dictionary;
002
003import java.io.Serializable;
004
005/**
006 * Specifies the necessary methods for functors that return whether nodes of
007 * some type represent the final states of an automaton.
008 * @author Dylon Edwards
009 * @param <State> Kind of the state this function accepts.
010 * @since 2.1.0
011 */
012public interface IFinalFunction<State> extends Serializable {
013
014  /**
015   * Whether the state represents the last character of some term.
016   * @param current State to check for finality.
017   * @return Whether this node represents the last character of some term.
018   */
019  boolean at(State current);
020}
