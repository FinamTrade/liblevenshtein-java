001package com.github.dylon.liblevenshtein.levenshtein;
002
003import java.io.Serializable;
004
005/**
006 * Returns a collection of spelling candidates for the given query term.
007 * @param <CandidateType> Kind of spelling candidate returned (e.g.
008 * {@link Candidate} or {@link java.lang.String}).
009 * @author Dylon Edwards
010 * @since 2.1.0
011 */
012public interface ITransducer<CandidateType> extends Serializable {
013
014  /**
015   * Finds all terms in the dictionary that are fewer than n units of spelling
016   * errors away from the query term.
017   * @param term Query term whose spelling candidates should bee determined
018   * @return A data structure consisting of all the spelling candidates for the
019   * query term.
020   */
021  ICandidateCollection<CandidateType> transduce(String term);
022
023  /**
024   * Finds all terms in the dictionary that are fewer than n units of spelling
025   * errors away from the query term.
026   * @param term Query term whose spelling candidates should bee determined
027   * @param maxDistance Maximum number of spelling errors the spelling
028   * candidates may have from the query term.
029   * @return A data structure consisting of all the spelling candidates for the
030   * query term.
031   */
032  ICandidateCollection<CandidateType> transduce(String term, int maxDistance);
033}
