001package com.github.dylon.liblevenshtein.levenshtein;
002
003/**
004 * Transitions a standard, Levenshtein position to all possible positions, given
005 * a set of parameters.
006 * @author Dylon Edwards
007 * @since 2.1.0
008 */
009public class StandardPositionTransitionFunction
010  extends AbstractPositionTransitionFunction {
011
012  private static final long serialVersionUID = 1L;
013
014  /**
015   * {@inheritDoc}
016   */
017  @Override
018  public IState of(
019      final int n,
020      final int[] position,
021      final boolean[] characteristicVector,
022      final int offset) {
023
024    final int i = position[0];
025    final int e = position[1];
026    final int h = i - offset;
027    final int w = characteristicVector.length;
028
029    // The edit distance is not below its maximum, allowed value.
030    if (e < n) {
031      // Consider any character before the last one of the spelling candidate
032      if (h <= w - 2) {
033        final int a = (n - e < Integer.MAX_VALUE)
034          ? n - e + 1
035          : Integer.MAX_VALUE;
036        final int b = w - h;
037        final int k = a < b ? a : b;
038        final int j = indexOf(characteristicVector, k, h);
039
040        if (0 == j) {
041          return stateFactory.build(
042              // [No Error]: Increment the index by one; leave the error alone.
043              positionFactory.build(1 + i, e)
044          );
045        }
046
047        if (j > 0) {
048          return stateFactory.build(
049              // [Insertion]: Leave the index alone; increment the error by one.
050              positionFactory.build(i, e + 1),
051              // [Substitution]: Increment both the index and error by one.
052              positionFactory.build(i + 1, e + 1),
053              // [Deletion]: Increment the index by one-more than the number of
054              // deletions; increment the error by the number of deletions.
055              positionFactory.build(i + j + 1, e + j)
056          );
057        }
058
059        // else, j < 0
060        return stateFactory.build(
061            // [Insertion]: Leave the index alone; increment the error by one.
062            positionFactory.build(i, e + 1),
063            // [Substitution]: Increment both the index and error by one.
064            positionFactory.build(i + 1, e + 1)
065        );
066      }
067
068      // Consider the last character of the spelling candidate
069      if (h == w - 1) {
070        if (characteristicVector[h]) {
071          return stateFactory.build(
072              // [No Error]: Increment the index by one; leave the error alone.
073              positionFactory.build(i + 1, e)
074          );
075        }
076
077        return stateFactory.build(
078            // [Insertion]: Leave the index alone; increment the error by one.
079            positionFactory.build(i, e + 1),
080            // [Substitution]: Increment both the index and error by one.
081            positionFactory.build(i + 1, e + 1)
082        );
083      }
084
085      // else, h == w
086      return stateFactory.build(
087          // [Insertion]: Leave the index alone; increment the error by one.
088          positionFactory.build(i, e + 1)
089      );
090    }
091
092    // The edit distance is at its maximum, allowed value.  Only consider this
093    // spelling candidate if there is no error at the index of its current term.
094    if ((e == n) && (h <= w - 1) && characteristicVector[h]) {
095      return stateFactory.build(
096          // [No Error]: Increment the index by one; leave the error alone.
097          positionFactory.build(i + 1, n)
098      );
099    }
100
101    // [Too Many Errors]: The edit distance has exceeded the max distance for
102    // the candidate term.
103    return null;
104  }
105}
