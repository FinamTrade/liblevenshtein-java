001package com.github.dylon.liblevenshtein.collection.dawg.factory;
002
003import java.io.Serializable;
004
005import com.github.dylon.liblevenshtein.collection.dawg.IDawgNode;
006import com.github.dylon.liblevenshtein.collection.dawg.Transition;
007
008/**
009 * Builds {@link Transition}s that link nodes together, under character labels.
010 * @param <NodeType> Kind of nodes linked by the tranistions.
011 * @author Dylon Edwards
012 * @since 2.1.0
013 */
014public class TransitionFactory<NodeType extends IDawgNode<NodeType>>
015  implements ITransitionFactory<NodeType>, Serializable {
016
017  private static final long serialVersionUID = 1L;
018
019  /**
020   * {@inheritDoc}
021   */
022  @Override
023  public Transition<NodeType> build(
024      final NodeType source,
025      final char label,
026      final NodeType target) {
027
028    return new Transition<NodeType>()
029      .source(source)
030      .label(label)
031      .target(target);
032  }
033}
