001package com.github.liblevenshtein.transducer;
002
003/**
004 * Transitions a merge-and-split, Levenshtein position to all possible
005 * positions, given a set of parameters.
006 * @author Dylon Edwards
007 * @since 2.1.0
008 */
009public class MergeAndSplitPositionTransitionFunction extends PositionTransitionFunction {
010
011  private static final long serialVersionUID = 1L;
012
013  /**
014   * {@inheritDoc}
015   */
016  @Override
017  public State of(
018      final int n,
019      final Position position,
020      final boolean[] characteristicVector,
021      final int offset) {
022
023    final int i = position.termIndex();
024    final int e = position.numErrors();
025    final boolean s = position.isSpecial();
026    final int h = i - offset;
027    final int w = characteristicVector.length;
028
029    if (e == 0 && 0 < n) {
030      if (h <= w - 2) {
031        if (characteristicVector[h]) {
032          return stateFactory.build(
033              // [No Error]: Increment index by one; leave error alone
034              positionFactory.build(i + 1, e, false)
035          );
036        }
037
038        return stateFactory.build(
039            // [Insertion]: Leave index alone; incrmeent error by one.
040            positionFactory.build(i, e + 1, false),
041            // [Split]: Leave index alone; increment error by one.
042            positionFactory.build(i, e + 1, true),
043            // [Substitution]: Increment index by one; increment error by one.
044            positionFactory.build(i + 1, e + 1, false),
045            // [Merge]: Increment index by two; increment error by one.
046            positionFactory.build(i + 2, e + 1, false)
047        );
048      }
049
050      if (h == w - 1) {
051        if (characteristicVector[h]) {
052          return stateFactory.build(
053              // [No Error]: Increment index by one; leave error alone.
054              positionFactory.build(i + 1, e, false)
055          );
056        }
057
058        return stateFactory.build(
059            // [Insertion]: Leave index alone; increment error by one.
060            positionFactory.build(i, e + 1, false),
061            // [Split]: Leave index alone; increment error by one.
062            positionFactory.build(i, e + 1, true),
063            // [Substitution]: Increment index by one; increment error by one.
064            positionFactory.build(i + 1, e + 1, false)
065        );
066      }
067
068      // else, h == w
069      return stateFactory.build(
070          // [Insertion]: Leave index alone; increment error by one.
071          positionFactory.build(i, e + 1, false)
072      );
073    }
074
075    if (e < n) {
076      if (h <= w - 2) {
077        if (!s) {
078          if (characteristicVector[h]) {
079            return stateFactory.build(
080                // [No Error]: Increment index by one; leave error alone.
081                positionFactory.build(i + 1, e, false)
082            );
083          }
084
085          return stateFactory.build(
086              // [Insertion]: Leave index alone; increment index by one.
087              positionFactory.build(i, e + 1, false),
088              // [Split]: Leave index alone; increment error by one.
089              positionFactory.build(i, e + 1, true),
090              // [Substitution]: Increment index by one; increment error by one.
091              positionFactory.build(i + 1, e + 1, false),
092              // [Merge]: Increment index by two; incremnt error by one.
093              positionFactory.build(i + 2, e + 1, false)
094          );
095        }
096
097        return stateFactory.build(
098            // [No Error]: Increment index by one; leave error alone.
099            positionFactory.build(i + 1, e, false)
100        );
101      }
102
103      if (h == w - 1) {
104        if (!s) {
105          if (characteristicVector[h]) {
106            return stateFactory.build(
107                // [No Error]: Increment index by one; leave error alone.
108                positionFactory.build(i + 1, e, false)
109            );
110          }
111
112          return stateFactory.build(
113              // [Insertion]: Leave index alone; increment error by one.
114              positionFactory.build(i, e + 1, false),
115              // [Split]: Leave index alone; increment error by one.
116              positionFactory.build(i, e + 1, true),
117              // [Substitution]: Increment index by one; increment error by one.
118              positionFactory.build(i + 1, e + 1, false)
119          );
120        }
121
122        return stateFactory.build(
123            // [No Error]: Increment index by one; leave error alone.
124            positionFactory.build(i + 1, e, false)
125        );
126      }
127
128      // else, h == w
129      return stateFactory.build(
130          // [Insertion]: Leave index alone; increment error by one.
131          positionFactory.build(i, e + 1, false)
132      );
133    }
134
135    if (h <= w - 1) {
136      if (!s) {
137        if (characteristicVector[h]) {
138          return stateFactory.build(
139              // [No Error]: Increment index by one; leave error alone.
140              positionFactory.build(i + 1, n, false)
141          );
142        }
143
144        return null;
145      }
146
147      return stateFactory.build(
148          // [No Error]: Increment index by one; leave error alone.
149          positionFactory.build(i + 1, e, false)
150      );
151    }
152
153    // else, h == w
154    // ------------
155    // [Too Many Errors]: The edit distance has exceeded the max distance for
156    // the candidate term.
157    return null;
158  }
159}
