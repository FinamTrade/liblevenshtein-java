001package com.github.liblevenshtein.transducer;
002
003/**
004 * Transitions a transposition, Levenshtein position to all possible positions,
005 * given a set of parameters.
006 * @author Dylon Edwards
007 * @since 2.1.0
008 */
009public class TranspositionPositionTransitionFunction extends PositionTransitionFunction {
010
011  private static final long serialVersionUID = 1L;
012
013  /**
014   * {@inheritDoc}
015   */
016  @Override
017  @SuppressWarnings("checkstyle:methodlength")
018  public State of(
019      final int n,
020      final Position position,
021      final boolean[] characteristicVector,
022      final int offset) {
023
024    final int i = position.termIndex();
025    final int e = position.numErrors();
026    final boolean t = position.isSpecial();
027    final int h = i - offset;
028    final int w = characteristicVector.length;
029
030    if (e == 0 && 0 < n) {
031      if (h <= w - 2) {
032        final int a = n - e < Integer.MAX_VALUE
033          ? n - e + 1
034          : Integer.MAX_VALUE;
035        final int b = w - h;
036        final int k = a < b ? a : b;
037        final int j = indexOf(characteristicVector, k, h);
038
039        switch (j) {
040          case 0:
041            return stateFactory.build(
042                // [No Error]: Increment the index by one; leave the error and
043                // special state alone
044                positionFactory.build(i + 1, 0, false)
045            );
046          case 1:
047            return stateFactory.build(
048                // [Insertion]: Leave index alone; increment the error by one.
049                positionFactory.build(i, 1, false),
050                // [Transposition]: Leave index alone; increment error by one.
051                positionFactory.build(i, 1, true),
052                // [Substitution]: Replace the current character; increment the
053                // index by one; increment the error by one.
054                positionFactory.build(i + 1, 1, false),
055                // [Deletion]: Increment the index by one-more than a single
056                // deletion; increment the error by one.
057                positionFactory.build(i + 2, 1, false)
058            );
059          case -1:
060            return stateFactory.build(
061                // [Insertion]: Leave index alone; increment the error by one.
062                positionFactory.build(i, 1, false),
063                // [Substitution]: Replace the current character; increment the
064                // index by one; increment the error by one.
065                positionFactory.build(i + 1, 1, false)
066            );
067          default: // j > 1
068            return stateFactory.build(
069                // [Insertion]: Leave index alone; increment the error by one.
070                positionFactory.build(i, 1, false),
071                // [Substitution]: Replace the current character; increment the
072                // index by one; increment the error by one.
073                positionFactory.build(i + 1, 1, false),
074                // [Deletion]: Increment index by one-more than the number of
075                // deletions; increment error by the number of deletions.
076                positionFactory.build(i + j + 1, j, false)
077            );
078        }
079      }
080
081      if (h == w - 1) {
082        if (characteristicVector[h]) {
083          return stateFactory.build(
084              // [No Error]: Increment index by one; leave error alone
085              positionFactory.build(i + 1, 0, false)
086          );
087        }
088
089        return stateFactory.build(
090            // [Insertion]: Leave index alone; increment error by one.
091            positionFactory.build(i, 1, false),
092            // [Substitution]: Increment index by one; increment error by one.
093            positionFactory.build(i + 1, 1, false)
094        );
095      }
096
097      // else, h == 2
098      return stateFactory.build(
099          // [Insertion]: Leave index alone; increment error by one.
100          positionFactory.build(i, 1, false)
101      );
102    }
103
104    if (1 <= e && e < n) {
105      if (h <= w - 2) {
106        if (!t) {
107          final int a = n - e < Integer.MAX_VALUE
108            ? n - e + 1
109            : Integer.MAX_VALUE;
110          final int b = w - h;
111          final int k = a < b ? a : b;
112          final int j = indexOf(characteristicVector, k, h);
113
114          switch (j) {
115            case 0:
116              return stateFactory.build(
117                  // [No Error]: Increment index by one; leave error alone.
118                  positionFactory.build(i + 1, e, false)
119              );
120            case 1:
121              return stateFactory.build(
122                  // [Insertion]: Leave index alone; increment error by one.
123                  positionFactory.build(i, e + 1, false),
124                  // [Transposition]: Leave index alone; increment error by one.
125                  positionFactory.build(i, e + 1, true),
126                  // [Substitution]: Increment index by one; increment error by
127                  // one.
128                  positionFactory.build(i + 1, e + 1, false),
129                  // [Deletion]: Increment index by one-more than one deletion;
130                  // increment error by one.
131                  positionFactory.build(i + 2, e + 1, false)
132              );
133            case -1:
134              return stateFactory.build(
135                  // [Insertion]: Leave index alone; increment error by one.
136                  positionFactory.build(i, e + 1, false),
137                  // [Substitution]: Increment index by one; increment error by
138                  // one.
139                  positionFactory.build(i + 1, e + 1, false)
140              );
141            default: // j > 1
142              return stateFactory.build(
143                  // [Insertion]: Leave index alone; increment error by one.
144                  positionFactory.build(i, e + 1, false),
145                  // [Substitution]: Increment index by one; increment error by
146                  // one.
147                  positionFactory.build(i + 1, e + 1, false),
148                  // [Deletion]: Increment index by one-more than number of
149                  // deletions; increment error by number of deletions.
150                  positionFactory.build(i + j + 1, e + j, false)
151              );
152          }
153        }
154
155        if (characteristicVector[h]) {
156          return stateFactory.build(
157              // [No Error]: Increment index by two; leave error alone.
158              positionFactory.build(i + 2, e, false)
159          );
160        }
161
162        // [Too Many Errors]: The edit distance has exceeded the max distance
163        // for the candidate term.
164        return null;
165      }
166
167      if (h == w - 1) {
168        if (characteristicVector[h]) {
169          return stateFactory.build(
170              // [No Error]: Increment index by one; leave error alone.
171              positionFactory.build(i + 1, e, false)
172          );
173        }
174
175        return stateFactory.build(
176            // [Insertion]: Leave index alone; increment error by one.
177            positionFactory.build(i, e + 1, false),
178            // [Substitution]: Increment index by one; leave error alone.
179            positionFactory.build(i + 1, e + 1, false)
180        );
181      }
182
183      // else, h == w
184      return stateFactory.build(
185          // [Insertion]: Leave index alone; increment error by one.
186          positionFactory.build(i, e + 1, false)
187      );
188    }
189
190    if (h <= w - 1 && !t) {
191      if (characteristicVector[h]) {
192        return stateFactory.build(
193            // [No Error]: Increment index by one; leave error alone.
194            positionFactory.build(i + 1, n, false)
195        );
196      }
197
198      // [Too Many Errors]: The edit distance has exceeded the max distance for
199      // the candidate term.
200      return null;
201    }
202
203    if (h <= w - 2 && t) {
204      if (characteristicVector[h]) {
205        return stateFactory.build(
206            // [No Error]: Increment index by two; leave error alone.
207            positionFactory.build(i + 2, n, false)
208        );
209      }
210
211      // [Too Many Errors]: The edit distance has exceeded the max distance for
212      // the candidate term.
213      return null;
214    }
215
216    // else, h == w
217    // ------------
218    // [Too Many Errors]: The edit distance has exceeded the max distance for
219    // the candidate term.
220    return null;
221  }
222}
