001package com.github.liblevenshtein.distance;
002
003import java.io.Serializable;
004
005/**
006 * Specifies the interface that all distance functions must implement.
007 * @param <Term> Type of the term whose distance is being determined.
008 * @author Dylon Edwards
009 * @since 2.1.0
010 */
011public interface IDistance<Term> extends Serializable {
012
013  /**
014   * Finds the distance between two terms, {@code v} and {@code w}. The distance
015   * between two terms is complemented by their similarity, which is determined
016   * by subtracting their distance from the maximum distance they may be apart.
017   * @param v Term to compare with {@code w}
018   * @param w Term to compare with {@code v}
019   * @return Distance between {@code v} and {@code w}
020   */
021  int between(Term v, Term w);
022}
