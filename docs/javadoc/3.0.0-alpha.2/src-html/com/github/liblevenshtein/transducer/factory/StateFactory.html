001package com.github.liblevenshtein.transducer.factory;
002
003import java.io.Serializable;
004
005import com.github.liblevenshtein.transducer.Position;
006import com.github.liblevenshtein.transducer.State;
007
008/**
009 * Builds Levenshtein states.
010 * @author Dylon Edwards
011 * @since 2.1.0
012 */
013public class StateFactory implements Serializable {
014
015  private static final long serialVersionUID = 1L;
016
017  /**
018   * Builds a new, Levenshtein state with the given position vectors.
019   * @param positions Array of position vectors to link into the state.
020   * @return New state having the position vectors.
021   */
022  public State build(final Position... positions) {
023    final State state = new State();
024
025    Position prev = null;
026    for (final Position curr : positions) {
027      state.insertAfter(prev, curr);
028      prev = curr;
029    }
030
031    return state;
032  }
033}
