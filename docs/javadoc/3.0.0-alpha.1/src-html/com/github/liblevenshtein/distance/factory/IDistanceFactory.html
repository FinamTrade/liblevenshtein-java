001package com.github.liblevenshtein.distance.factory;
002
003import java.io.Serializable;
004
005import com.github.liblevenshtein.transducer.Algorithm;
006import com.github.liblevenshtein.distance.IDistance;
007
008/**
009 * Builds instances of Levenshtein distance metrics.
010 * @param <Term> Type of the terms whose distances are measurable.
011 * @author Dylon Edwards
012 * @since 2.1.0
013 */
014public interface IDistanceFactory<Term> extends Serializable {
015
016  /**
017   * Returns a Levenshtein distance metric that utilizes the request algorithm.
018   * @param algorithm Type of metric that should be returned
019   * @return A distance metric that utilizes the request alagorithm
020   */
021  IDistance<Term> build(Algorithm algorithm);
022}
