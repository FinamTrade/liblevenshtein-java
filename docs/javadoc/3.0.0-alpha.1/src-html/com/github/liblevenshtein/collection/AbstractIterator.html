001package com.github.liblevenshtein.collection;
002
003import java.util.Iterator;
004
005/**
006 * Implements boilerplate logic I typically use in my iterators.
007 * @param <Type> Type of the elements returned from this iterator.
008 * @author Dylon Edwards
009 * @since 2.1.0
010 */
011public abstract class AbstractIterator<Type> implements Iterator<Type> {
012
013  /**
014   * Next value to return. The next element to return is determined via
015   * {@link #advance()}
016   */
017  protected Type next = null;
018
019  /**
020   * {@inheritDoc}
021   */
022  @Override
023  public boolean hasNext() {
024    advance();
025    return next != null;
026  }
027
028  /**
029   * {@inheritDoc}
030   */
031  @Override
032  public Type next() {
033    advance();
034    final Type nextLocal = this.next;
035    this.next = null;
036    return nextLocal;
037  }
038
039  /**
040   * {@inheritDoc}
041   */
042  @Override
043  public void remove() {
044    throw new UnsupportedOperationException(
045        "java.util.Iterator.remove() is not supported");
046  }
047
048  /**
049   * Advances {@link #next} to the next, non-null value to return.
050   */
051  protected abstract void advance();
052}
