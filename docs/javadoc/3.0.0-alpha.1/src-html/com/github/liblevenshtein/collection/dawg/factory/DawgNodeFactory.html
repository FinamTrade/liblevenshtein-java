001package com.github.liblevenshtein.collection.dawg.factory;
002
003import java.io.Serializable;
004
005import it.unimi.dsi.fastutil.chars.Char2ObjectMap;
006import it.unimi.dsi.fastutil.chars.Char2ObjectRBTreeMap;
007
008import com.github.liblevenshtein.collection.dawg.DawgNode;
009import com.github.liblevenshtein.collection.dawg.FinalDawgNode;
010
011/**
012 * Builds nodes for use in DAWG structures. This implementation uses an object
013 * pool to avoid unnecessary memory consumption and garbage collection.
014 *
015 * @author Dylon Edwards
016 * @since 2.1.0
017 */
018public class DawgNodeFactory implements IDawgNodeFactory<DawgNode>, Serializable {
019
020  private static final long serialVersionUID = 1L;
021
022  /**
023   * {@inheritDoc}
024   */
025  @Override
026  public DawgNode build() {
027    return build(false);
028  }
029
030  /**
031   * {@inheritDoc}
032   */
033  @Override
034  public DawgNode build(final boolean isFinal) {
035    final Char2ObjectMap<DawgNode> edges = new Char2ObjectRBTreeMap<>();
036
037    if (isFinal) {
038      return new FinalDawgNode(edges);
039    }
040
041    return new DawgNode(edges);
042  }
043}
