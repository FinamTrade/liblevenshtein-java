001package com.github.liblevenshtein.collection.dawg.factory;
002
003import java.io.Serializable;
004
005import com.github.liblevenshtein.collection.dawg.IDawgNode;
006import com.github.liblevenshtein.collection.dawg.Transition;
007
008/**
009 * Builds transitions that link nodes together, connected by a character label.
010 * @param <NodeType> Kind of the nodes joined by transitions.
011 * @author Dylon Edwards
012 * @since 2.1.0
013 */
014public interface ITransitionFactory<NodeType extends IDawgNode<NodeType>> extends Serializable {
015
016  long serialVersionUID = 1L;
017
018  /**
019   * Builds or recycles a {@link Transition} object representing an edge from
020   * {@code source} to {@code target}, annotated with {@code label}.
021   * @param source Node from which the transition is leaving.
022   * @param label Label that joins {@code source} and {@code target} together.
023   * @param target Node to which the transition is going.
024   * @return A {@link Transition} linking {@code source} to {@code target},
025   * under {@code label}.
026   */
027  Transition<NodeType> build(NodeType source, char label, NodeType target);
028
029  /**
030   * Recycles a {@link Transition} to be re-used from
031   * {@link #build(IDawgNode,char,IDawgNode) build}.
032   * @param transition {@link Transition} to recycle.  Note that once the
033   * transition has been recycled, you should discard it (like a deleted
034   * reference in C++).
035   */
036  default void recycle(final Transition<NodeType> transition) {
037    // default behavior is to do nothing
038  }
039}
