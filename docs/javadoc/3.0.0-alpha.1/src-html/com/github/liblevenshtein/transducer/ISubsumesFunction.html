001package com.github.liblevenshtein.transducer;
002
003import java.io.Serializable;
004
005/**
006 * Determines whether the position represented by the combination of parameters
007 * in the left half subsumes the position represented by the combination of
008 * parameters in the right half.
009 * @author Dylon Edwards
010 * @since 2.1.0
011 */
012public interface ISubsumesFunction extends Serializable {
013
014  /**
015   * Determines whether the standard, Levenshtein state represented by
016   * {@code (i,e)} subsumes the other, represented by {@code (j,f)}.
017   * @param i Index of the spelling candidate of the first position.
018   * @param e Number of accumulated errors of the spelling candidate at index
019   * {@code i} of the first position.
020   * @param j Index of the spelling candidate of the second position.
021   * @param f Number of accumulated errors of the spelling candidate at index
022   * {@code j} of the second position.
023   * @return Whether {@code (i,e)} subsumes {@code (j,f)}
024   */
025  boolean at(int i, int e, int j, int f);
026
027  /**
028   * Determines whether the transposition or merge-and-split, Levenshtein state
029   * represented by {@code (i,e,s)} subsumes the other, represented by
030   * {@code (j,f,t)}, under the maximum number of errors, {@code n}.
031   * @param i Index of the spelling candidate of the first position.
032   * @param e Number of accumulated errors of the spelling candidate at index
033   * {@code i} of the first position.
034   * @param s Either {@code 0} or {@code 1}, depending on whether the first
035   * position is a regular state or a special case (e.g. transposition, merge,
036   * or split).
037   * @param j Index of the spelling candidate of the second position.
038   * @param f Number of accumulated errors of the spelling candidate at index
039   * {@code j} of the second position.
040   * @param t Either {@code 0} or {@code 1}, depending on whether the second
041   * position is a regular state or a special case (e.g. transposition, merge,
042   * or split).
043   * @param n Maximum number of errors allowed in spelling candidates.
044   * @return Whether {@code (i,e,s)} subsumes {@code (j,f,t)}
045   */
046  boolean at(int i, int e, int s, int j, int f, int t, int n);
047}
