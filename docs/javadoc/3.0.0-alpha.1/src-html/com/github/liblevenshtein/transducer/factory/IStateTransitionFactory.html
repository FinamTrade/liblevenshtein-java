001package com.github.liblevenshtein.transducer.factory;
002
003import java.io.Serializable;
004
005import com.github.liblevenshtein.transducer.IStateTransitionFunction;
006
007/**
008 * Builds functions that transition states to other states, given some evidence.
009 * @author Dylon Edwards
010 * @since 2.1.0
011 */
012public interface IStateTransitionFactory extends Serializable {
013
014  long serialVersionUID = 1L;
015
016  /**
017   * Builds a new state-transition function that only considers spelling
018   * candidates no more than {@code maxDistance} errors from the query term.
019   * @param maxDistance Maximum number of errors tolerated in transforming the
020   * query term into the spelling candidate.
021   * @return New state-transition function that only considers spelling
022   * candidates within {@code maxDistance} errors from the query term.
023   */
024  IStateTransitionFunction build(int maxDistance);
025
026  /**
027   * Recycles a state-transition function so it may be re-used in
028   * {@link #build(int) build}.
029   * @param transition {@link IStateTransitionFunction} to recycle.  You must
030   * discard your reference to {@code transition} once you've recycled it.
031   */
032  default void recycle(final IStateTransitionFunction transition) {
033    // default behavior is to do nothing
034  }
035}
