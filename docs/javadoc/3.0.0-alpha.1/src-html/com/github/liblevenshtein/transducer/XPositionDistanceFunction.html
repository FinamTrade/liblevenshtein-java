001package com.github.liblevenshtein.transducer;
002
003import java.io.Serializable;
004
005/**
006 * Distance function for the Levenshtein distance algorithms, extended with
007 * transposition (standard- and t-positions), and with merge and split
008 * (standard- and s-positions). Because this functor applies to both the
009 * transposition and merge-and-split automata, it uses the identifier, x, to
010 * represent both t and s.
011 * @author Dylon Edwards
012 * @since 2.1.0
013 */
014public class XPositionDistanceFunction implements IDistanceFunction, Serializable {
015
016  private static final long serialVersionUID = 1L;
017
018  /**
019   * {@inheritDoc}
020   */
021  @Override
022  public int at(final IState state, final int w) {
023    int minimumDistance = Integer.MAX_VALUE;
024
025    for (int m = 0; m < state.size(); ++m) {
026      final int[] position = state.getOuter(m);
027
028      // (1 == x) -> t-position or s-position
029      // (0 == x) -> standard position
030      // only calculate the distance for standard positions.
031      final int x = position[2];
032
033      if (0 == x) {
034        final int i = position[0];
035        final int e = position[1];
036        final int distance = w - i + e;
037        if (distance < minimumDistance) {
038          minimumDistance = distance;
039        }
040      }
041    }
042
043    return minimumDistance;
044  }
045}
