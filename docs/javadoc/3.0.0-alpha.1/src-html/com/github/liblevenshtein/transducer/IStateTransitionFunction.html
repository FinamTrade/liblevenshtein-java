001package com.github.liblevenshtein.transducer;
002
003import java.io.Serializable;
004
005/**
006 * Transitions Levenshtein states according to the evidence.
007 * @author Dylon Edwards
008 * @since 2.1.0
009 */
010public interface IStateTransitionFunction extends Serializable {
011
012  /**
013   * Returns the state consisting of all the possible position-transitions from
014   * the current state, given the characteristic vector.
015   * @param state Source state to transition from.
016   * @param characteristicVector Relevant subwords denoting whether the indices
017   * of the current, spelling candidate from the dictionary has characters
018   * matching the one being sought from the query term.
019   * @return Next state, consisting of all possible position-transitions from
020   * the source {@code state}, conditioned on {@code characteristicVector}.
021   */
022  IState of(IState state, boolean[] characteristicVector);
023}
