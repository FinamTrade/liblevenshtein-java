128  public synchronized boolean add(@NonNull final String term) {
129    if (term == null) {
130      throw new java.lang.IllegalArgumentException("term is null");
131    }
132    if (term.compareTo(previousTerm) < 0) {
133      throw new IllegalArgumentException("Due to caveats with the current DAWG implementation, terms must be inserted in ascending order");
134    }
135    // Special Case: Empty String
136    if (term.isEmpty()) {
137      root = factory.build(true);
138      return true;
139    }
140    final int upperBound = (term.length() < previousTerm.length()) ? term.length() : previousTerm.length();
141    // Find the length of the longest, common prefix between term and
142    // previousTerm
143    int i = 0;
144    while (i < upperBound && term.charAt(i) == previousTerm.charAt(i)) {
145      i += 1;
146    }
147    // Check the unchecked nodes for redundancy, proceeding from the last one
148    // down to the common prefix size. Then, truncate the list at that point.
149    minimize(i);
150    // Add the suffix, starting from the correct node, mid-way through the graph
151    DawgNode node = (null == uncheckedTransitions.peekFirst()) ? root : uncheckedTransitions.peekFirst().target();
152    for (int k = term.length() - 1; i < k; i += 1) {
153      final char label = term.charAt(i);
154      final DawgNode nextNode = factory.build();
155      uncheckedTransitions.addFirst(transitionFactory.build(node, label, nextNode));
156      node = nextNode;
157    }
158    if (i < term.length()) {
159      final char label = term.charAt(i);
160      final DawgNode nextNode = factory.build(true);
161      uncheckedTransitions.addFirst(transitionFactory.build(node, label, nextNode));
162    }
163    //node = nextNode;
164    //i += 1;
165    previousTerm = term;
166    size += 1;
167    return true;
168  }

177  /**
178   * Builds this DAWG in such a way that it remains a minimal trie.
179   * @param lowerBound Number of pending transitions to leave for the next
180   * round (they will be the most-recent transitions).
181   */
182  private void minimize(final int lowerBound) {
183    // Proceed from the leaf up to a certain point
184    for (int j = uncheckedTransitions.size(); j > lowerBound; j -= 1) {
185      final Transition<DawgNode> transition = uncheckedTransitions.removeFirst();
186      final DawgNode source = transition.source();
187      final char label = transition.label();
188      final DawgNode target = transition.target();
189      final boolean isFinal = target.isFinal();
190      final DawgNode existing = minimizedNodes.get(target);
191      if (null != existing) {
192        source.addEdge(label, existing);
193      } else {
194        source.addEdge(label, target);
195        minimizedNodes.put(target, target);
196      }
197    }
