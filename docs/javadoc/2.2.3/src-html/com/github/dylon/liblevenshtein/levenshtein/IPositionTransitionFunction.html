001package com.github.dylon.liblevenshtein.levenshtein;
002
003import java.io.Serializable;
004
005/**
006 * Transitions an input position of a Levenshtein state to all candidate
007 * positions, conditioned by a set of parameters.
008 * @author Dylon Edwards
009 * @since 2.1.0
010 */
011public interface IPositionTransitionFunction extends Serializable {
012
013  /**
014   * Returns a state with all the possible transitions from the current
015   * position.
016   * @param n Maximum number of errors to tolerate in spelling candidates.
017   * @param position Vector consisting of the current index of the spelling
018   * candidate, the number of errors accumulated up to that index, and
019   * (optionally) a flag specifying whether the position is a special kind as
020   * defined by its specific, Levenshtein algorithm (e.g. whether it is a
021   * transposition position, etc.).
022   * @param characteristicVector Relevant subwords consisting of booleans about
023   * whether the characters associated by their indices in the spelling
024   * candidate are the same as the character being sought from the query term.
025   * @param offset Offset for various operations within the transition function.
026   * @return New state consisting of all possible transitions for the given
027   * position, given the other paramters.
028   */
029  IState of(int n, int[] position, boolean[] characteristicVector, int offset);
030}
