001package com.github.dylon.liblevenshtein.levenshtein.factory;
002
003import java.io.Serializable;
004import java.util.Collection;
005
006import com.github.dylon.liblevenshtein.levenshtein.Algorithm;
007import com.github.dylon.liblevenshtein.levenshtein.ITransducer;
008
009/**
010 * Fluent builder of Levenshtein transducers.
011 * @author Dylon Edwards
012 * @since 2.1.0
013 */
014public interface ITransducerBuilder extends Serializable {
015
016  /**
017   * Specifies the collection of dictionary terms for the dictionary automaton.
018   * This method assumes the collection is not sorted.
019   * @param dictionary Collection of dictionary terms to consider when
020   * generating spelling candidates.
021   * @return This {@link ITransducerBuilder} or an equivalent one, for fluency.
022   */
023  ITransducerBuilder dictionary(Collection<String> dictionary);
024
025  /**
026   * Specifies the collection of dictionary terms for the dictionary automaton.
027   * @param dictionary Collection of dictionary terms to consider when
028   * generating spelling candidates.
029   * @param isSorted Whether the dictionary is sorted.  If it is not sorted then
030   * it will probably be sorted.
031   * @return This {@link ITransducerBuilder} or an equivalent one, for fluency.
032   */
033  ITransducerBuilder dictionary(Collection<String> dictionary, boolean isSorted);
034
035  /**
036   * Specifies what Levenshtein algorithm to use for the transducer.
037   * @param algorithm Levenshtein algorithm to use for the transducer.
038   * @return This {@link ITransducerBuilder} or an equivalent one, for fluency.
039   */
040  ITransducerBuilder algorithm(Algorithm algorithm);
041
042  /**
043   * Specifies the default, maximum tolerated error of spelling candidates from
044   * the query term.  This value is used if the transducer is called without
045   * explicitly specifying the maximum distance.  It default to infinity, which
046   * will return every term in the dictionary, so if you plan on using the
047   * implicit maximum distance then you should set it here.
048   * @param defaultMaxDistance Default, maximum tolerated errors between
049   * spelling candidates and the query term.
050   * @return This {@link ITransducerBuilder} or an equivalent one, for fluency.
051   */
052  ITransducerBuilder defaultMaxDistance(int defaultMaxDistance);
053
054  /**
055   * Specifies whether the distance between each spelling candidate and the
056   * query term should be returned with the spelling candidate.
057   * @param includeDistance Whether to include distances with spelling
058   * candidates.
059   * @return This {@link ITransducerBuilder} or an equivalent one, for fluency.
060   */
061  ITransducerBuilder includeDistance(boolean includeDistance);
062
063  /**
064   * If you only desire a subset of the spelling candidates for a query term,
065   * you may specify the number of candidates to return, here.  Note that the
066   * candidates are not guaranteed to be ordered such that you would get the
067   * nearest candidates in the collection if you limit its size.
068   * @param maxCandidates Maximum number of candidates to include in the
069   * collections of spelling candidates.
070   * @return This {@link ITransducerBuilder} or an equivalent one, for fluency.
071   */
072  ITransducerBuilder maxCandidates(int maxCandidates);
073
074  /**
075   * Builds a Levenshtein transducer according to the parameters set for this
076   * {@link ITransducerBuilder}.
077   * @param <CandidateType> Implicit type of the spelling candidates generated
078   * by the transducer.
079   * @return Levenshtein transducer for seeking spelling candidates for query
080   * terms (fuzzy searching!).
081   */
082  <CandidateType> ITransducer<CandidateType> build();
083}
