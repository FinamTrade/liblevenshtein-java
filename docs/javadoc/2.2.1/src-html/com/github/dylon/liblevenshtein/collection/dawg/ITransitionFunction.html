001package com.github.dylon.liblevenshtein.collection.dawg;
002
003import java.io.Serializable;
004
005import it.unimi.dsi.fastutil.chars.CharIterator;
006
007/**
008 * Deterministically-transitions between states according to some input.
009 * @param <State> Kind of state this transition function accepts.
010 * @author Dylon Edwards
011 * @since 2.1.0
012 */
013public interface ITransitionFunction<State> extends Serializable {
014
015  /**
016   * Determines the next state given the current one and some input.
017   * @param current The active state of an automaton
018   * @param label Input used to determine the next state
019   * @return The next state of the automaton
020   */
021  State of(State current, char label);
022
023  /**
024   * Returns the labels of the outgoing edges from the node.
025   * @param current The active state of an automaton
026   * @return The labels of the outgoing edges from the node.
027   */
028  CharIterator of(State current);
029}
