001package com.github.dylon.liblevenshtein.collection.dawg.factory;
002
003import java.io.Serializable;
004
005import com.github.dylon.liblevenshtein.collection.dawg.Prefix;
006
007/**
008 * Builds instances of {@link Prefix}.
009 * @param <DictionaryNode> Kind of node enclosed by the prefixes.
010 * @author Dylon Edwards
011 * @since 2.1.0
012 */
013public interface IPrefixFactory<DictionaryNode> extends Serializable {
014
015  long serialVersionUID = 1L;
016
017  /**
018   * Builds or recycles a {@link Prefix} instance, representing {@code node} and
019   * the string built by traversing its trie from the root to itself, collecting
020   * the characters along the way.
021   * @param node Node that the {@link Prefix} represents.
022   * @param value String representing the prefix of the term, from the root to
023   * {@code node}.
024   * @return A {@link Prefix} representing {@code node} and containing the
025   * prefix of the term from the root to {@code node}.
026   */
027  Prefix<DictionaryNode> build(DictionaryNode node, String value);
028
029  /**
030   * Recycles a {@link Prefix} to be re-used in {@link #build(Object,String) build}.
031   * @param prefix The {@link Prefix} to recycle.  Note that once you recycle
032   * {@code prefix}, you should discard it.
033   */
034  default void recycle(final Prefix<DictionaryNode> prefix) {
035    // default behavior is to do nothing
036  }
037}
