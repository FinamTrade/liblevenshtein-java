001package com.github.dylon.liblevenshtein.levenshtein;
002
003import java.io.Serializable;
004
005/**
006 * The distance of each position in a state can be defined as follows:
007 * <p>
008 * <code>distance = w - i + e</code>
009 * <p>
010 * For every accepting position, it must be the case that <code>w - i &le; n - e</code>.
011 * It follows directly that the distance of every accepted position must be no
012 * more than {@code n}:
013 * <p>
014 * <code>(w - i &le; n - e) &hArr; (w - i + e &le; n) &hArr; (distance &le; n)</code>
015 * <p>
016 * The Levenshtein distance between any two terms is defined as the minimum
017 * edit distance between the two terms.  Therefore, iterate over each position
018 * in an accepting state, and take the minimum distance among all its accepting
019 * positions as the corresponding Levenshtein distance.
020 * @author Dylon Edwards
021 * @since 2.1.0
022 */
023public interface IDistanceFunction extends Serializable {
024
025  /**
026   * @param state Levenshtein state whose minimum, Levenshtein distance is to be
027   * determined.
028   * @param w Length of the query term.
029   * @return Minimum distance between the spelling candidate and query term,
030   * implied by the positions within {@code state}.
031   */
032  int at(IState state, int w);
033}
