001package com.github.dylon.liblevenshtein.levenshtein;
002
003import java.io.Serializable;
004import java.util.Comparator;
005
006/**
007 * Defines the methods on Levenshtein states, which consist of all possible
008 * position vectors at the current index of the spelling candidate.
009 * @author Dylon Edwards
010 * @since 2.1.0
011 */
012public interface IState extends Serializable {
013
014  /**
015   * Number of positions in this state.
016   * @return Number of positions in this state.
017   */
018  int size();
019
020  /**
021   * Adds a new position to the linked-list of positions in this state.
022   * @param position Position to add to those already in this state.
023   */
024  void add(int[] position);
025
026  /**
027   * Inserts a position into a specific location of the linked-list of positions.
028   * @param index Location at which to insert the position.
029   * @param position Position vector to insert into this state.
030   */
031  void insert(int index, int[] position);
032
033  /**
034   * One of two cursors maintained in this state, which returns the position at
035   * the requested index.  Subsequent requests for positions at the immediately
036   * proceeding or preceding indices will begin at the index of the last
037   * position requested, so lookup for consecutive positions is fast.
038   * <p>
039   * {@link #getOuter(int)} is used within the outer loops of
040   * {@link com.github.dylon.liblevenshtein.levenshtein.UnsubsumeFunction}s.
041   * @param index Location of the position to return
042   * @return Position vector at the requested index.
043   * @see #getInner(int)
044   */
045  int[] getOuter(int index);
046
047  /**
048   * One of two cursors maintained in this state, which returns the position at
049   * the requested index.  Subsequent requests for positions at the immediately
050   * proceeding or preceding indices will begin at the index of the last
051   * position requested, so lookup for consecutive positions is fast.
052   * <p>
053   * {@link #getInner(int)} is used within the inner loops of
054   * {@link com.github.dylon.liblevenshtein.levenshtein.UnsubsumeFunction}s.
055   * @param index Location of the position to return
056   * @return Position vector at the requested index.
057   * @see #getOuter(int)
058   */
059  int[] getInner(int index);
060
061  /**
062   * Unlinks the position corresponding to the inner cursor, at the last index
063   * requested of {@link #getInner(int)}.
064   * @return Position vector that was unlinked from those in this state.
065   */
066  int[] removeInner();
067
068  /**
069   * Unlinks and recycles all the positions in this state, making it empty. This
070   * is usually done just before this state is itself recycled.
071   */
072  void clear();
073
074  /**
075   * Merge-sorts the positions in this state in a fashion that makes
076   * un-subsumption easy.
077   * @param comparator Describes how to sort the positions (dependent on the
078   * Levenshtein algorithm).
079   */
080  void sort(Comparator<int[]> comparator);
081}
